347  [[nodiscard]] std::chrono::nanoseconds stdvExecTimeCluster() const {
348  double ret = 0;
349  if (this->isInCluster()) {
350  auto mean = this->meanExecTimeCluster();
351  for (auto it = this->belongingNode()->insideNodes()->equal_range(this->coreClusterNode()).first;
352  it != this->belongingNode()->insideNodes()->equal_range(this->coreClusterNode()).second; ++it) {
353  ret += std::pow(it->second->executionTime().count() - mean.count(), 2);
354  }
355  ret /= (this->numberThreads() - 1);
356  ret = std::sqrt(ret);
357  }
358  return std::chrono::nanoseconds((int64_t)(ret));
359  }
