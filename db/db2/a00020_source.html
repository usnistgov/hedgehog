226  void recycleMemory(std::shared_ptr<MemoryData<ManagedMemory>> managedMemory) {
227  std::lock_guard<std::mutex> lk(memoryManagerMutex_);
228  managedMemory->used();
229  if (managedMemory->canBeRecycled()) {
230  this->profiler_->addReleaseMarker();
231  managedMemory->recycle();
232  this->pool_->push_back(std::dynamic_pointer_cast<ManagedMemory>(managedMemory));
233  }
