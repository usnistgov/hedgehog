144  void createCluster(std::shared_ptr<std::multimap<CoreNode *, std::shared_ptr<CoreNode>>> &insideNodesGraph) override {
145  auto mm = this->task()->memoryManager();
146  // Set cluster property
147  if (this->numberThreads() > 1) { this->setInCluster(); }
148 
149  // For each copy
150  for (size_t threadId = 1; threadId < this->numberThreads(); threadId++) {
151  // Copy the task
152  auto taskCopy = createCopyFromThis();
153  // Duplicate the memory manager, each of them will have a separate instance
154  connectMemoryManager(mm, taskCopy);
155  // Set property
156  auto coreCopy = std::dynamic_pointer_cast<CoreTask<TaskOutput, TaskInputs...>>(taskCopy->core());
157  coreCopy->threadId(threadId);
158  coreCopy->coreClusterNode(this);
159  coreCopy->copyInnerStructure(this);
160  coreCopy->setInCluster();
161  insideNodesGraph->insert({this, coreCopy});
162  }

166  void run() override {
167  HLOG_SELF(2, "Run")
168 #ifndef HH_DISABLE_PROFILE
169  std::chrono::time_point<std::chrono::system_clock>
170  start,
171  finish;
172 #endif
173 
174  this->isActive(true);
175 #ifndef HH_DISABLE_NVTX_PROFILE
176  this->nvtxProfiler()->initialize(this->threadId());
177 #endif
178  // Do the initialization phase
179  this->preRun();
180 
181  // If automatic start is enable send nullptr to all input nodes and wake them up
182  if (this->automaticStart()) {
183 #ifndef HH_DISABLE_PROFILE
184  start = std::chrono::system_clock::now();
185 #endif
186  (static_cast<CoreExecute<TaskInputs> *>(this)->callExecute(nullptr), ...);
187 #ifndef HH_DISABLE_PROFILE
188  finish = std::chrono::system_clock::now();
189  this->incrementExecutionDuration(std::chrono::duration_cast<std::chrono::nanoseconds>(finish - start));
190 #endif
191  }
192 
193  // Actual computation loop
194  while (!this->callCanTerminate(true)) {
195 #ifndef HH_DISABLE_PROFILE
196  start = std::chrono::system_clock::now();
197 #endif
198  // Wait for a data to arrive or termination
199  volatile bool canTerminate = this->waitForNotification();
200 #ifndef HH_DISABLE_PROFILE
201  finish = std::chrono::system_clock::now();
202  this->incrementWaitDuration(std::chrono::duration_cast<std::chrono::nanoseconds>(finish - start));
203 #endif
204  // If can terminate break the loop early
205  if (canTerminate) { break; }
206 #ifndef HH_DISABLE_PROFILE
207  start = std::chrono::system_clock::now();
208 #endif
209  // Operate the receivers to get a data and send it to execute
210  (this->operateReceiver<TaskInputs>(), ...);
211 #ifndef HH_DISABLE_PROFILE
212  finish = std::chrono::system_clock::now();
213  this->incrementExecutionDuration(std::chrono::duration_cast<std::chrono::nanoseconds>(finish - start));
214 #endif
215  }
216 
217  // Do the shutdown phase
218  this->postRun();
219  // Wake up node that this is linked to
220  this->wakeUp();
221  }

226  virtual bool callCanTerminate(bool lock) {
227  bool result;
228  if (lock) { this->lockUniqueMutex(); }
229  result = this->node()->canTerminate();
230  HLOG_SELF(2, "callCanTerminate: " << std::boolalpha << result)
231  if (lock) { this->unlockUniqueMutex(); }
232  return result;
233  };

237  template<class Input>
238  void operateReceiver() {
239  HLOG_SELF(2, "Operate receivers")
240  // Lock the mutex
241  this->lockUniqueMutex();
242  // Get the receiver with the right type
243  auto receiver = static_cast<CoreQueueReceiver<Input> *>(this);
244  // If receiver's queue not empty
245  if (!receiver->receiverEmpty()) {
246  // Get the data
247  std::shared_ptr<Input> data = receiver->popFront();
248  this->incrementNumberReceivedElements();
249  this->unlockUniqueMutex();
250  // Call execute
251  std::chrono::time_point<std::chrono::system_clock> start = std::chrono::system_clock::now();
252  static_cast<CoreExecute<Input> *>(this)->callExecute(data);
253  std::chrono::time_point<std::chrono::system_clock> finish = std::chrono::system_clock::now();
254  this->incrementPerElementExecutionDuration(std::chrono::duration_cast<std::chrono::nanoseconds>(finish - start));
255  } else {
256  // Unlock the mutex
257  this->unlockUniqueMutex();
258  }
259  }

264  bool waitForNotification() override {
265 #ifndef HH_DISABLE_NVTX_PROFILE
266  this->nvtxProfiler()->startRangeWaiting(this->totalQueueSize());
267 #endif
268  std::unique_lock<std::mutex> lock(*(this->slotMutex()));
269  HLOG_SELF(2, "Wait for notification")
270  this->notifyConditionVariable()->wait(lock,
271  [this]() {
272  bool receiversEmpty = this->receiversEmpty();
273  bool callCanTerminate = this->callCanTerminate(false);
274  HLOG_SELF(2,
275  "Check for notification: " << std::boolalpha
276  << (bool) (!receiversEmpty) << "||"
277  << std::boolalpha
278  << (bool) callCanTerminate)
279  return !receiversEmpty || callCanTerminate;
280  });
281  HLOG_SELF(2, "Notification received")
282 #ifndef HH_DISABLE_NVTX_PROFILE
283  this->nvtxProfiler()->endRangeWaiting();
284 #endif
285  return callCanTerminate(false);
286  }

290  std::shared_ptr<AbstractTask<TaskOutput, TaskInputs...>> createCopyFromThis() {
291  HLOG_SELF(0, "Copy Whole Task")
292 
293  std::shared_ptr<AbstractTask<TaskOutput, TaskInputs...>> sharedAbstractTaskCopy = this->task()->copy();
294  // The copy method has not been redefined and return nullptr by default
295  if (!sharedAbstractTaskCopy) {
296  std::ostringstream oss;
297  oss << "A copy for the task \"" << this->name()
298  << "\" has been invoked but return nullptr. To fix this error, overload the AbstractTask::copy function and "
299  "return a valid object.";
300  HLOG_SELF(0, oss.str())
301  throw (std::runtime_error(oss.str()));
302  }
303 
304  // Add the copy into the cluster
305  this->clusterAbstractTask_.push_back(sharedAbstractTaskCopy);
306 
307  // Get the core from the copy
308  auto coreCopy = std::dynamic_pointer_cast<CoreTask<TaskOutput, TaskInputs...>>(sharedAbstractTaskCopy->core());
309 
310  if (coreCopy == nullptr) {
311  std::ostringstream oss;
312  oss << "A copy for the task " << this->name()
313  << " has been invoked but the AbstractTask constructor has not been called. To fix this error, call the "
314  "AbstractTask constructor in your specialized task constructor.";
315  HLOG_SELF(0, oss.str())
316  throw (std::runtime_error(oss.str()));
317  }
318 
319  // Copy the property
320  coreCopy->automaticStart_ = this->automaticStart();
321  coreCopy->threadId(this->threadId());
322  coreCopy->isInside(true);
323  if (this->isInCluster()) { coreCopy->setInCluster(); }
324  coreCopy->numberThreads(this->numberThreads());
325 
326  // Copy the memory manager
327  if (this->task()->memoryManager()) {
328  auto copyMemoryManager = this->task()->memoryManager()->copy();
329  if (!copyMemoryManager) {
330  std::ostringstream oss;
331  oss << "An execution pipeline has been created with a graph that hold a task named \""
332  << this->name()
333  << "\" connected to a memory manager. Or the memory manager does not have a compliant copy method. "
334  "Please implement it.";
335  HLOG_SELF(0, oss.str())
336  throw (std::runtime_error(oss.str()));
337  }
338  HLOG_SELF(0, "Copy the MM " << this->task()->memoryManager() << " to: " << copyMemoryManager
339  << " and set it to the task: " << coreCopy->name() << " / " << coreCopy->id())
340  connectMemoryManager(copyMemoryManager, sharedAbstractTaskCopy);
341  }
342  return sharedAbstractTaskCopy;
343  }
