98  void initialize() final {
99  int numGpus = 0;
100  int canAccess = 0;
101  checkCudaErrors(cudaGetDeviceCount(&numGpus));
102  assert(this->deviceId() < numGpus);
103  checkCudaErrors(cudaSetDevice(this->deviceId()));
104  checkCudaErrors(cudaStreamCreate(&stream_));
105 
106  if (enablePeerAccess_) {
107  for (int i = 0; i < numGpus; ++i) {
108  if (i != this->deviceId()) {
109  checkCudaErrors(cudaDeviceCanAccessPeer(&canAccess, this->deviceId(), i));
110 
111  if (canAccess) {
112  auto ret = cudaDeviceEnablePeerAccess(i, 0);
113  if (ret != cudaErrorPeerAccessAlreadyEnabled) {
114  checkCudaErrors(ret);
115  }
116  peerDeviceIds_.insert(i);
117  }
118  }
119  }
120  }
121  auto ret = cudaGetLastError();
122  if (ret != cudaErrorPeerAccessAlreadyEnabled) {
123  checkCudaErrors(ret);
124  }
125  this->initializeCuda();
126  }
127 
131  void shutdown() final {
132  this->shutdownCuda();
133  checkCudaErrors(cudaStreamDestroy(stream_));
134  }
135 
137  virtual void initializeCuda() {}
138 
140  virtual void shutdownCuda() {}
141 
144  bool enablePeerAccess() const { return enablePeerAccess_; }
145 
148  cudaStream_t stream() const { return stream_; }
149 
153  bool hasPeerAccess(int peerDeviceId) { return peerDeviceIds_.find(peerDeviceId) != peerDeviceIds_.end(); }
154 };
