89 if (!this->isInitialized()) {
90 this->initialized();
91 std::for_each(

94 emptyShared = std::make_shared<T>();
95 emptyShared->memoryManager(this);
96 }
97 );
98 initializeMemoryManager();
99 }
100 memoryManagerMutex_.unlock();
101 }
