304  bool isMemoryManaged) final {
305  std::ostringstream
306  oss;
307 
308  std::string
309  headLabel,
310  tailLabel,
311  idDest,
312  queueStr,
313  penWidth = ",penwidth=1";
314 
315  // Change the width of the edge if the data is memory managed
316  if (isMemoryManaged) { penWidth = ",penwidth=3"; }
317 
318  // If the queue information need to be printed, add queue information
319  if (this->structureOptions_ == StructureOptions::QUEUE || this->structureOptions_ == StructureOptions::ALL) {
320  oss << " QS:" << queueSize << " MQS:" << maxQueueSize;
321  queueStr = oss.str();
322  oss.str("");
323  }
324 
325  // If all nodes in a cluster need to be printed

333  // If the destination node is in a cluster

339  idDest = "box" + dest.second;
340  oss << source.first << " -> " << idDest << "[label=\"" << edgeType << queueStr << "\"" << headLabel
341  << tailLabel << penWidth << "];";
342  }
343  // If the destination node is NOT in a cluster
344  } else {
345  oss << source.first << " -> " << to->id() << "[label=\"" << edgeType << queueStr << "\"" << headLabel
346  << penWidth
347  << "];";
348  }
349  }
350  // If the source node is NOT in a cluster
351  } else {
352  // If the destination node is in a cluster
353  if (to->isInCluster()) {
354  // Loop over all destination node of the current node
355  // (could be simply itself or, all input nodes in the case of graph)
356  for (auto &dest : to->ids()) {
357  tailLabel = ",lhead=cluster" + dest.second;
358  idDest = "box" + dest.second;
359  oss << from->id() << " -> " << idDest << "[label=\"" << edgeType << queueStr << "\"" << headLabel
360  << tailLabel << penWidth << "];";
361  }
362  // If the destination node is NOT in a cluster
363  } else {
364  oss << from->id() << " -> " << to->id() << "[label=\"" << edgeType << queueStr << "\"" << headLabel
365  << penWidth
366  << "];";
367  }
368  }
369  edges_.push_back(oss.str());
370  // If all nodes should NOT be printed and, from and to are the main nodes of their cluster
371  } else if (from->id() == from->coreClusterNode()->id() && to->id() == to->coreClusterNode()->id()) {
372  oss << from->id() << " -> " << to->id() << "[label=\"" << edgeType << queueStr << "\"" << penWidth << "];";

381  std::string getNodeInformation(core::CoreNode *node) {
382  std::stringstream ss;
383 
384  // Print the name
385  ss << node->id() << " [label=\"" << node->name();
386  // Print the id (adress) in case of debug
387  if (debugOptions_ == DebugOptions::ALL) {
388  ss << " " << node->id() << " \\(" << node->threadId() << ", " << node->graphId() << "\\)";
389  }
390 
391  switch (node->type()) {
392  // Set a specific shape for the source
393  case core::NodeType::Source:ss << "\", shape=doublecircle";
394  break;
395  // Set a specific shape for the sink
396  case core::NodeType::Sink:ss << "\",shape=point";
397  break;
398  // For the execution task
399  case core::NodeType::Task:
400  // If the cluster has to be presented as a single dot node
401  if (!(this->structureOptions_ == StructureOptions::ALLTHREADING
402  || this->structureOptions_ == StructureOptions::ALL)) {
403  if (node->isInCluster()) {
404  ss << " x " << node->numberThreads();

408  if (debugOptions_ == DebugOptions::ALL) {
409  // Print number of active input connection

412  if (this->structureOptions_ == StructureOptions::ALLTHREADING
413  || this->structureOptions_ == StructureOptions::ALL) {
414  ss << "\\nThread Active?: " << std::boolalpha << dynamic_cast<core::CoreSlot *>(node)->isActive();

416  } else {
417  ss << "\\nActive threads: " << dynamic_cast<core::CoreSlot *>(node)->numberActiveThreadInCluster();

581  static std::string durationPrinter(std::chrono::nanoseconds const &ns) {
582  std::ostringstream oss;
583 
584  // Cast with precision loss
585  auto s = std::chrono::duration_cast<std::chrono::seconds>(ns);
586  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(ns);
587  auto us = std::chrono::duration_cast<std::chrono::microseconds>(ns);
588 
589  if (s > std::chrono::seconds::zero()) {
590  oss << s.count() << "." << std::setfill('0') << std::setw(3) << (ms - s).count() << "s";
591  } else if (ms > std::chrono::milliseconds::zero()) {
592  oss << ms.count() << "." << std::setfill('0') << std::setw(3) << (us - ms).count() << "ms";
593  } else if (us > std::chrono::microseconds::zero()) {
594  oss << us.count() << "." << std::setfill('0') << std::setw(3) << (ns - us).count() << "us";
595  } else {
596  oss << std::setw(3) << ns.count() << "ns";
597  }
598  return oss.str();
599  }
