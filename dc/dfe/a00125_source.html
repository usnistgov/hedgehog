283  void addEdge(std::shared_ptr<UserDefinedSender> from, std::shared_ptr<UserDefinedMultiReceiver> to) {
284  assert(from != nullptr && to != nullptr);
285  static_assert(traits::Contains_v<Output, Inputs>, "The given Receiver cannot be linked to this Sender");
286  if (this->isInside()) {
287  std::ostringstream oss;
288  oss << "You can not modify a graph that is connected inside another graph: " << __FUNCTION__;
289  HLOG_SELF(0, oss.str())
290  throw (std::runtime_error(oss.str()));
291  }
292 
293  //Get the associated cores
294  auto coreSender = dynamic_cast<CoreSender <Output> *>(std::static_pointer_cast<behavior::Node>(from)->core().get());
295  auto coreNotifier = dynamic_cast<CoreNotifier *>(coreSender);
296  auto coreSlot = dynamic_cast<CoreSlot *>(std::static_pointer_cast<behavior::Node>(to)->core().get());
297  auto
298  coreReceiver = dynamic_cast<CoreReceiver<Output> *>(std::static_pointer_cast<behavior::Node>(to)->core().get());
299 
300  if (from->core().get() == this || to->core().get() == this) {
301  std::ostringstream oss;
302  oss << "You can not connect a graph to itself: " << __FUNCTION__;
303  HLOG_SELF(0, oss.str())
304  throw (std::runtime_error(oss.str()));
305  }
306 
307  if (coreSender->hasBeenRegistered()) {
308  if (coreSender->belongingNode() != this) {
309  std::ostringstream oss;
310  oss << "The Sender node should belong to the graph: " << __FUNCTION__;
311  HLOG_SELF(0, oss.str())
312  throw (std::runtime_error(oss.str()));
313  }
314  }
315 
316  if (coreReceiver->hasBeenRegistered()) {
317  if (coreReceiver->belongingNode() != this) {
318  std::ostringstream oss;
319  oss << "The Receiver node should belong to the graph: " << __FUNCTION__;
320  HLOG_SELF(0, oss.str())
321  throw (std::runtime_error(oss.str()));
322  }
323  }
324 
325  HLOG_SELF(0,
326  "Add edge from " << coreSender->name() << "(" << coreSender->id() << ") to " << coreReceiver->name()
327  << "(" << coreReceiver->id()
328  << ")")
329 
330  for (auto r : coreReceiver->receivers()) { coreSender->addReceiver(r); }
331  for (CoreSlot *slot : coreSlot->getSlots()) { coreNotifier->addSlot(slot); }
332  for (auto s : coreSender->getSenders()) {
333  coreReceiver->addSender(s);
334  coreSlot->addNotifier(s);
335  }
336 
337  this->registerNode(std::dynamic_pointer_cast<CoreNode>(from->core()));
338  this->registerNode(std::dynamic_pointer_cast<CoreNode>(to->core()));
339  }

795  void duplicateInsideNodes(CoreGraph<GraphOutput, GraphInputs...> const &rhs) {
796  // Inside nodes to copy, MainClusterNode -> ClusterNodes
797  std::multimap<CoreNode *, std::shared_ptr<CoreNode>> &originalInsideNodes = *(rhs.insideNodes());
798  // Correspondence map, original node -> copy node
799  std::map<CoreNode *, std::shared_ptr<CoreNode>> correspondenceMap;
800  //Duplicate node
801  std::shared_ptr<CoreNode> duplicate;
802 
803  // Create all the duplicates and link them to their original node
804  for (std::pair<CoreNode *const, std::shared_ptr<CoreNode>> const &originalNode : originalInsideNodes) {
805  duplicate = originalNode.second->clone();
806  duplicate->belongingNode(this);
807  correspondenceMap.insert({originalNode.second.get(), duplicate});
808  }
809 
810  // Add the duplicate node into the insideNode structure
811  for (std::pair<CoreNode *const, std::shared_ptr<CoreNode>> const &originalNode : originalInsideNodes) {
812  // Original node
813  CoreNode *originalInsideNode = originalNode.second.get();
814  // Copy node
815  std::shared_ptr<CoreNode> duplicateInsideNode = correspondenceMap.find(originalInsideNode)->second;
816  duplicateInsideNode->belongingNode(this);
817  this->insideNodes()->insert({duplicateInsideNode.get(), duplicateInsideNode});
818  }
819 
820  //Do the linkage
821  for (std::pair<CoreNode *const, std::shared_ptr<CoreNode>> const &originalNode : originalInsideNodes) {
822  CoreNode *originalInsideNode = originalNode.second.get();
823  std::shared_ptr<CoreNode> duplicateInsideNode = correspondenceMap.find(originalInsideNode)->second;
824  originalInsideNode->duplicateEdge(duplicateInsideNode.get(), correspondenceMap);
825  }
826 
827  // Duplicate input nodes
828  for (CoreNode *originalInputNode : *(rhs.inputsCoreNodes())) {
829  auto shInputCoreNode = correspondenceMap.find(originalInputNode)->second;
830  auto inputCoreNode = shInputCoreNode.get();
831  this->inputsCoreNodes_->insert(inputCoreNode);
832  (this->duplicateInputNodes<GraphInputs>(dynamic_cast<CoreReceiver<GraphInputs> *>(inputCoreNode)), ...);
833  this->registerNode(shInputCoreNode);
834  }
835 
836  // Duplicate output nodes
837  for (CoreSender <GraphOutput> *originalOutputNode : *(rhs.outputCoreNodes())) {
838  auto shOutputCoreNode = correspondenceMap.find(originalOutputNode)->second;
839  if (auto outputCoreNode = dynamic_cast<CoreSender <GraphOutput> *>(shOutputCoreNode.get())) {
840  this->outputCoreNodes_->insert(outputCoreNode);
841 
842  for (CoreSender <GraphOutput> *sender : outputCoreNode->getSenders()) {
843  this->sink_->addNotifier(sender);
844  this->sink_->addSender(sender);
845  }
846 
847  outputCoreNode->addSlot(this->sink_.get());
848  outputCoreNode->addReceiver(this->sink_.get());
849 
850  this->registerNode(std::static_pointer_cast<CoreNode>(shOutputCoreNode));
851  } else {
852  std::ostringstream oss;
853  oss << "Internal error, the output node is not a sender: " << __FUNCTION__;
854  HLOG_SELF(0, oss.str())
855  throw (std::runtime_error(oss.str()));
856  }
857  }
858  }
